Loop variable reference in goroutines
Cause: 

func main() {
    data := []int{1, 2, 3, 4, 5}

    var wg sync.WaitGroup
    wg.Add(len(data))

    for _, v := range data { // loop ends before goroutines execute
        go func() {
            defer wg.Done()
            fmt.Println(v)  // closures capturing and evaluating same reference to `v`
        }()
    }

    wg.Wait()
}
Loop variable reference in goroutines
Common fixes: 

    for _, v := range data {
        go func(v int) {
            defer wg.Done()
            fmt.Println(v)
        }(v) // `v` evaluated and value captured here before dispatching goroutine
    }
or 

    for _, v := range data {
        v := v // `v` evaluated and value captured in block-scoped variable `v`
        go func() {
            defer wg.Done()
            fmt.Println(v)
        }()
    }

Provides a way to send a "stop" or "done" signal to a goroutine:

... with the context package

Core pattern (playground ): 

func foo() interface{} {
    return "Hello, World!"
}

func main() { // main goroutine
    ctx, cancel := context.WithCancel(context.Background()) // or context.WithDeadline()
    defer cancel()                                          // cancel when we're done with the context

    result := make(chan interface{}) // setup result channel

    go func() { // sub goroutine
        select { // leverage `select` to handle cancellation or result, whichever comes first
        case <-ctx.Done():
            fmt.Println("cancelled")
        default:
            result <- foo()
        }
    }()

    select { // leverage `select` to handle timeout or result, whichever comes first
    case r := <-result:
        fmt.Println(r)
    case <-time.After(time.Second):
        panic("timeout") // handle timeout
    }

    // cancel() is always invoked
}

func main() {
    search := func(ctx context.Context, query string, results chan<- string) {
        max, min := 10_000, 1000
        delayMillis := rand.Intn(max-min) + min

        select {
        case <-ctx.Done():
            fmt.Println("search cancelled")
        case <-time.After(time.Duration(delayMillis) * time.Millisecond): // work simulation
            results <- fmt.Sprintf("results for '%s'", query)
        }
    }

    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    query := "hello world!"
    results := make(chan string)

    for i := 0; i < 10; i++ {
        go search(ctx, query, results)
    }

    result := <-results

    cancel()                          // unnecessary, but good for demo purposes
    time.Sleep(50 * time.Millisecond) // unnecessary, but good for demo purposes

    fmt.Println(result)
}
